window.onload = function() {
    // Parse tags from :tag: format and set data attributes
    document.querySelectorAll('.outline-2 p, .outline-4 p').forEach(function(p) {
        var text = p.innerHTML;
        var tagMatch = text.match(/:(\w+):/g);
        if (tagMatch) {
            var tags = tagMatch.map(function(t) { return t.replace(/:/g, ''); });
            p.classList.add('article-item');
            p.dataset.tags = tags.join(' ');
            // Remove the :tag: text from display
            p.innerHTML = text.replace(/\s*:(\w+):/g, '');
        } else {
            p.classList.add('article-item');
            p.dataset.tags = '';
        }
    });

    // Tag filtering
    var activeTags = new Set();
    
    window.toggleTag = function(tag, btn) {
        if (activeTags.has(tag)) {
            activeTags.delete(tag);
            btn.classList.remove('active');
        } else {
            activeTags.add(tag);
            btn.classList.add('active');
        }
        filterArticles();
    }
    
    function filterArticles() {
        var items = document.querySelectorAll('.article-item');
        var headings = document.querySelectorAll('.outline-2 h2, .outline-4 h4');
        
        if (activeTags.size === 0) {
            // No filters: show everything
            items.forEach(function(item) { item.classList.remove('hidden'); });
            headings.forEach(function(h) { h.classList.remove('hidden'); });
            return;
        }
        
        // Hide all headings when filtering
        headings.forEach(function(h) { h.classList.add('hidden'); });
        
        // Show only matching articles
        items.forEach(function(item) {
            var itemTags = (item.dataset.tags || '').split(' ');
            var match = false;
            activeTags.forEach(function(tag) {
                if (itemTags.includes(tag)) match = true;
            });
            item.classList.toggle('hidden', !match);
        });
    }

    // Original column layout code
    var firstHr = document.querySelector('hr');
    if (firstHr) {
        var div = document.createElement('div');
        div.className = 'outline-col';
        var parentDiv = firstHr.parentNode;
        parentDiv.parentNode.insertBefore(div, parentDiv.nextSibling);
        var outline2Elements = document.querySelectorAll('.outline-2');
        outline2Elements.forEach(function(el) {
            div.appendChild(el);
        });
    }

    // Dark mode functionality
    window.toggleDarkMode = function() {
        document.body.classList.toggle('dark-mode');
        const isDarkMode = document.body.classList.contains('dark-mode');
        localStorage.setItem('darkMode', isDarkMode);
        updateToggleButton();
    }

    function updateToggleButton() {
        const toggle = document.querySelector('.dark-mode-toggle-nav');
        if (toggle) {
            toggle.innerHTML = document.body.classList.contains('dark-mode') ? '‚òÄÔ∏è' : 'üåô';
        }
    }

    // Initialize dark mode from localStorage
    const savedDarkMode = localStorage.getItem('darkMode');
    if (savedDarkMode === 'true') {
        document.body.classList.add('dark-mode');
    }

    // Update toggle button on page load
    updateToggleButton();
};
